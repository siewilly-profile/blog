# f579. 1. 購物車
## 解題
我們利用兩個數字去紀錄出現的次數，只要有加入就在對應的位置+1，若拿出則在對應的位置上-1

## AC Code
### python
```python
from sys import stdin
a,b=map(int,stdin.readline().strip().split())
n=int(stdin.readline().strip())
ans=0
for _ in range(n):
    num=[0,0]
    temp=list(map(int,stdin.readline().strip().split()))
    for i in temp:
        if i==a: num[0]+=1
        if i==-1*a: num[0]-=1
        if i==b: num[1]+=1
        if i==-1*b: num[1]-=1
    if num[0]>0 and num[1]>0:
        ans+=1
print(ans)
```
### C++
```cpp
#include<bits/stdc++.h>
using namespace std;
int main(){
    int a,b,ans=0;
    cin>>a>>b;
    int n;
    cin>>n;
    for(int i=0;i<n;i++){
        int s1=0,s2=0;
        int input;
        while(true){
            cin>>input;
            if(input==0) break;
            if(input==a) s1++;
            if(input==-1*a) s1--;
            if(input==b) s2++;
            if(input==-1*b) s2--;
        }
        if(s1>0 && s2>0){
            ans++;
        }
    }
    cout<<ans<<endl;
    return 0;
}
```
